/*

Em C#, não existe tal método para remover ou adicionar elementos a uma array existente. É por isso que é recomendado usar uma lista ao invés de uma array. Mas podemos utilizar a cláusula where() do LINQ para encontrar o índice do elemento a ser removido e pular o elemento

*/



using System;
using System.Linq;

class StringToFloat {
  static void Main(string[] args) {
    string[] myArray = {
      "a",
      "b",
      "c",
      "d",
      "e"
    };
    Console.WriteLine("Array before deletion");
    foreach(string value in myArray) {
      Console.WriteLine(value);
    }
    int indexToRemove = 3;
    myArray = myArray.Where((source, index) =>index != indexToRemove).ToArray();
    Console.WriteLine("Array after deletion");

    foreach(string value in myArray) {
      Console.WriteLine(value);
    }
  }
}
